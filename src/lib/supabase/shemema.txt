CREATE TABLE public.assistance (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  assistance_date timestamp with time zone NOT NULL DEFAULT now(),
  customer_id uuid NOT NULL,
  CONSTRAINT assistance_pkey PRIMARY KEY (id),
  CONSTRAINT assistance_customer_id_fkey FOREIGN KEY (customer_id) REFERENCES public.customers(id)
);
CREATE TABLE public.customer_membership (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  customer_id uuid NOT NULL UNIQUE,
  membership_type character varying NOT NULL,
  last_payment_date timestamp with time zone,
  expiration_date timestamp with time zone,
  CONSTRAINT customer_membership_pkey PRIMARY KEY (id),
  CONSTRAINT customer_membership_customer_id_fkey FOREIGN KEY (customer_id) REFERENCES public.customers(id),
  CONSTRAINT customer_membership_membership_type_fkey FOREIGN KEY (membership_type) REFERENCES public.types_memberships(type)
);
CREATE TABLE public.customers (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  first_name character varying NOT NULL,
  last_name character varying NOT NULL,
  person_id character varying NOT NULL,
  phone character varying,
  email character varying,
  assistance_count integer DEFAULT 0,
  CONSTRAINT customers_pkey PRIMARY KEY (id)
);
CREATE TABLE public.profile (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  auth_id uuid NOT NULL DEFAULT auth.uid(),
  first_name text,
  last_name text,
  picture character varying,
  CONSTRAINT profile_pkey PRIMARY KEY (id),
  CONSTRAINT profile_auth_id_fkey FOREIGN KEY (auth_id) REFERENCES auth.users(id)
);
CREATE TABLE public.types_memberships (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  type character varying NOT NULL DEFAULT ''::character varying UNIQUE,
  CONSTRAINT types_memberships_pkey PRIMARY KEY (id)
);
